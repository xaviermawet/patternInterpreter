<expression> ::= <terme> '+' <expression>
               | <terme> '-' <expression>
               | <terme>

<terme> ::= <facteur> '*' <terme>
          | <facteur> '/' <terme>
          | <facteur>

<facteur> ::= <Atome> '^' <facteur>
            | <Atome>

<Atome> ::= <nombre>
          | '-' <atome>
          | '(' <expression> ')'
          | sin <atome>
          | cos <atome>
